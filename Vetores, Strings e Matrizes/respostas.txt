#include <stdio.h>

int main(){
	int valores[5], n, i;
	
	printf("Quantia em R$: ");
	scanf("%d", &n);
	if(n!=0){
		valores[0] = n/100;
		n %= 100;
	
		valores[1] = n/50;
		n%= 50;
	
		valores[2] = n/10;
		n%= 10;
	
		valores[3] = n/5;
		n%= 5;
	
		valores[4] = n/1;
			
		}
		
	else
		return 0;
	
	for(i=0; i<5; i++){
		if(valores[i]!=0){
			
			switch(i){
		
			case 0: 
				printf("%d cedulas de R$100\n\n", valores[0]);
				break;
			
			case 1:
				printf("%d cedulas de R$50\n\n", valores[1]);
				break;
			
			case 2: 
				printf("%d cedulas de R$10\n\n", valores[2]);
				break;
			
			case 3: 
				printf("%d cedulas de R$5\n\n", valores[3]);
				break;
			
			case 4: 
				printf("%d cedulas de R$1\n\n", valores[4]);
				break;
			}
			
		}
	
	
	}
}
